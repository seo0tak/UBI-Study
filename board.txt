------------------------------freeboard자유게시판

drop table free_board; 

--order by ref, re_level, num순서로 select

create table free_board(
num number primary key,	--free_board_seq.nextval
ref number not null,		--ref엔 free_board_seq.currval을 삽입.	--댓글 : 글을 달 대상의 ref를 삽입.
re_level number not null,	--기본으로 0이 들어간다.		--댓글 : 글을 달 대상의 re_level에 1을 더하여 삽입.
read_count number not null,	--조회수. 기본으로 0이 들어간다.
name varchar2(50),		--작성자
title varchar2(50),		--제목
content varchar2(50),		--내용
pw varchar2(50) not null,	--패스워드
day date default sysdate	--작성일자
);

drop sequence free_board_seq;
create sequence free_board_seq
minvalue 0
start with 0
nocache;
